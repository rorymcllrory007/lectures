from manimlib import *

class SquareToCircle(Scene):
    def construct(self):
        # 添加一个方块
        square = Square()
        self.play(ShowCreation(square))
        # 方块变为圆
        circle = Circle()
        self.play(Transform(square, circle))
        # 圆放大
        self.play(ApplyMethod(circle.scale,2))
        # 圆变回方块
        self.play(Transform(circle, square))
